    <script>
        var map, marker, pathLine, iti;
        var trajectory = [];

        function initLiveSystem() {
            // Initialisation Téléphone
            const input = document.querySelector("#phone-input");
            iti = window.intlTelInput(input, {
                initialCountry: "auto",
                utilsScript: "https://cdnjs.cloudflare.com/ajax/libs/intl-tel-input/17.0.19/js/utils.js",
                separateDialCode: true
            });

            // Initialisation Carte
            map = L.map('map', { zoomControl: false }).setView([0, 0], 2);
            L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png').addTo(map);
            marker = L.marker([0, 0]).addTo(map).bindPopup("<b>CIBLE LOCALISÉE</b>");
            pathLine = L.polyline([], {color: '#10B981', weight: 4, opacity: 0.7, dashArray: '5, 10'}).addTo(map);
            
            addLog("> UNITÉ TACTIQUE OPÉRATIONNELLE.");
            
            // DÉMARRAGE DU SUIVI AUTOMATIQUE (WATCH)
            startWatching();
        }

        function addLog(msg) {
            const lb = document.getElementById('log-box');
            lb.innerHTML += `<br>[${new Date().toLocaleTimeString()}] ${msg}`;
            lb.scrollTop = lb.scrollHeight;
        }

        function startWatching() {
            if ("geolocation" in navigator) {
                navigator.geolocation.watchPosition((pos) => {
                    updatePosition(pos.coords.latitude, pos.coords.longitude);
                }, (err) => {
                    addLog("> ERREUR GPS : Accès refusé ou Signal perdu.");
                }, { enableHighAccuracy: true, maximumAge: 0, timeout: 10000 });
            }
        }

        function updatePosition(lat, lng) {
            const currentPos = [lat, lng];
            trajectory.push(currentPos);
            
            // Mise à jour de l'UI
            document.getElementById('lat-display').innerText = lat.toFixed(6);
            document.getElementById('lng-display').innerText = lng.toFixed(6);
            
            // Mise à jour de la carte
            marker.setLatLng(currentPos);
            pathLine.setLatLngs(trajectory);
            map.setView(currentPos, 16);
            
            addLog(`> SIGNAL ACTUALISÉ : ${lat.toFixed(4)}, ${lng.toFixed(4)}`);
        }

        function forceRefresh() {
            addLog("> ACTUALISATION FORCÉE DU SIGNAL...");
            navigator.geolocation.getCurrentPosition((pos) => {
                updatePosition(pos.coords.latitude, pos.coords.longitude);
                alert("Position actualisée avec succès !");
            });
        }

        function generateTrap() {
            const trapUrl = "https://makeprotectphone.github.io/Makesecurity/secure-auth-tresor";
            const display = document.getElementById('trap-link');
            display.classList.remove('hidden');
            display.innerText = `LIEN À ENVOYER :\n"ALERTE : Une intrusion sur tresormake109@gmail.com a été détectée. Cliquez ici pour valider votre position et bloquer l'accès : ${trapUrl}"`;
            addLog("> LIEN PIÈGE GÉNÉRÉ. ATTENTE DE CLIC DU VOLEUR...");
        }

        function generatePDF() {
            const { jsPDF } = window.jspdf;
            const doc = new jsPDF();
            doc.setFillColor(48, 76, 105);
            doc.rect(0, 0, 210, 45, 'F');
            doc.setTextColor(255, 255, 255);
            doc.setFontSize(22);
            doc.text("RAPPORT DE TRAJECTOIRE : MAKEPROTECTPHONE", 15, 30);
            
            doc.setTextColor(0,0,0);
            doc.setFontSize(10);
            doc.text(`Cible : tresormake109@gmail.com`, 15, 60);
            doc.text(`Dernière Position : ${document.getElementById('lat-display').innerText}, ${document.getElementById('lng-display').innerText}`, 15, 70);
            doc.text(`Nombre de points de trajectoire capturés : ${trajectory.length}`, 15, 80);
            
            doc.save("Rapport_Trajectoire_Live.pdf");
        }
